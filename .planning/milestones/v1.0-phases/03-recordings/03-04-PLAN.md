    ```csharp
    private CancellationTokenSource? _archiveCts;

    [RelayCommand]
    private void CancelArchive()
    {
        _archiveCts?.Cancel();
    }

    [RelayCommand]
    private async Task ArchiveRecording()
    {
        _archiveCts = new CancellationTokenSource();
        IsArchiving = true;
        try
        {
            var progress = new Progress<double>(p => ArchiveProgress = p);
            if (IsTrip && Trip is not null)
                await _archiveService.ArchiveTripAsync(Trip, progress, _archiveCts.Token);
            else if (CurrentRecording is not null)
                await _archiveService.ArchiveAsync(CurrentRecording, progress, _archiveCts.Token);
            IsArchived = true;
        }
        catch (OperationCanceledException) { /* cancelled by user */ }
        finally
        {
            IsArchiving = false;
            _archiveCts?.Dispose();
            _archiveCts = null;
        }
    }
    ```

    **Create `src/BlackBoxBuddy/Services/DesktopMediaPlayerService.cs`:**
    LibVLCSharp implementation of IMediaPlayerService.
    ```csharp
    using LibVLCSharp.Shared;
    using BlackBoxBuddy.Services;
    namespace BlackBoxBuddy.Services;
    public class DesktopMediaPlayerService : IMediaPlayerService
    {
        private readonly LibVLC _libVlc;

        public DesktopMediaPlayerService()
        {
            Core.Initialize(); // Required LibVLCSharp initialization
            _libVlc = new LibVLC("--no-video-title-show");
        }

        public object CreatePlayer() => new MediaPlayer(_libVlc);

        public void Play(object player, Uri mediaUri)
        {
            if (player is MediaPlayer mp)
            {
                var media = new Media(_libVlc, mediaUri);
                mp.Media = media;
                mp.Play();
            }
        }

        public void Pause(object player)
        {
            if (player is MediaPlayer mp) mp.Pause();
        }

        public void Stop(object player)
        {
            if (player is MediaPlayer mp) mp.Stop();
        }

        public void Seek(object player, float position)
        {
            if (player is MediaPlayer mp) mp.Position = position;
        }

        public void SetRate(object player, float rate)
        {
            if (player is MediaPlayer mp) mp.SetRate(rate);
        }

        public void NextFrame(object player)
        {
            if (player is MediaPlayer mp) mp.NextFrame();
        }

        public void PreviousFrame(object player)
        {
            // LibVLC does not have a PreviousFrame API. Implement as seek back ~33ms (one frame at 30fps)
            if (player is MediaPlayer mp && mp.Time > 33)
                mp.Time -= 33;
        }

        public void DisposePlayer(object player)
        {
            if (player is MediaPlayer mp) mp.Dispose();
        }

        public void Dispose() => _libVlc.Dispose();
    }
    ```

    **Note:** `Core.Initialize()` must be called once before any LibVLC usage. Calling it in the constructor of the singleton service ensures it runs exactly once.
